import selenium
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.common.exceptions import NoSuchElementException

# Set up the webdriver
driver = webdriver.Chrome()

# Navigate to the website
driver.get("https://cymbal-shops.retail.cymbal.dev")

try:
    # Find and click on a random product link
    product_links = driver.find_elements(By.CSS_SELECTOR, ".hot-product-card a")
    import random
    random_product = random.choice(product_links)
    random_product.click()
except NoSuchElementException:
    print("Product link not found")

try:
    # Add the item to the cart
    add_to_cart = driver.find_element(By.XPATH, "//button[@id='add-to-cart']")
    add_to_cart.click()
except NoSuchElementException:
    print("Add to cart button not found")

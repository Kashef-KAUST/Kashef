```python
from selenium import webdriver
from selenium.webdriver.chrome.service import Service
from selenium.webdriver.common.by import By
from webdriver_manager.chrome import ChromeDriverManager
import time
import random

# Setup Chrome webdriver
service = Service(ChromeDriverManager().install())
driver = webdriver.Chrome(service=service)

# Navigate to the website
driver.get("https://cymbal-shops.retail.cymbal.dev")

# Allow some time for the page to load
time.sleep(5)

# Find all product links
product_links = driver.find_elements(By.CSS_SELECTOR, "div.hot-product-card a")

# Select a random product link
random_product_link = random.choice(product_links)

# Click on the random product link
random_product_link.click()

# Allow some time for the product page to load
time.sleep(5)

# Print the HTML content of the product page
print(driver.page_source)

# Close the browser
driver.quit()
```

This code selects a random product from the homepage by finding all elements that match the CSS selector for product links, choosing one at random, clicking on it, and then printing the HTML content of the product detail page.

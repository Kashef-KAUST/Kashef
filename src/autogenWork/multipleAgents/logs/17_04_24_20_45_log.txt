    # Click on the date field before sending keys
    date_field = driver.find_element(By.ID, "travel_booking_date")

    date_field.send_keys(future_date)
except Exception as e:

# Attempt to click the search button
try:
    driver.find_element(By.ID, "travel_searching_button").click()

# Handle potential alert

    alert = Alert(driver)
    alert.accept()  # Accept the alert
    print("Alert accepted.")

    print(f"No alert to accept: {e}")

# Print the HTML content of the page after attempting search
print(driver.page_source)

# Close the browser
driver.quit()

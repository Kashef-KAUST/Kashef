```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.chrome.service import Service
from webdriver_manager.chrome import ChromeDriverManager
import random
import time

# Initialize the WebDriver
driver = webdriver.Chrome(service=Service(ChromeDriverManager().install()))

# Navigate to the website
driver.get("https://cymbal-shops.retail.cymbal.dev")

# Wait for the page to load
time.sleep(5)  # It's better to use explicit waits, but for simplicity, we use sleep here.

# Find all product cards on the homepage
product_cards = driver.find_elements(By.CSS_SELECTOR, ".hot-product-card a")

# Select a random product card
random_product = random.choice(product_cards)

# Click on the selected product card to navigate to the product detail page
random_product.click()

# Print the current URL to confirm navigation
print(driver.current_url)

# Print the HTML content of the product detail page
print(driver.page_source)

# Close the browser
driver.quit()
```

1. Open the website.
2. Find all product cards on the homepage.
3. Select one product card at random.
4. Click on the selected product card to navigate to the product detail page.
5. Print the current URL and the HTML content of the product detail page to confirm successful navigation.
6. Close the browser.

```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.chrome.service import Service
from webdriver_manager.chrome import ChromeDriverManager
import time
import random

try:
    # Setup Chrome WebDriver
    driver = webdriver.Chrome(service=Service(ChromeDriverManager().install()))

    # Open the website
    driver.get("https://cymbal-shops.retail.cymbal.dev")

    # Wait for the page to load
    time.sleep(5)

    # Find all elements that are links to products
    product_links = driver.find_elements(By.CSS_SELECTOR, ".hot-product-card a")

    # Select a random product link and click on it
    random.choice(product_links).click()

    # Wait for the product detail page to load
    time.sleep(5)

    # Print the current URL and HTML content of the product detail page
    print("Current URL:", driver.current_url)
    print("HTML Content:", driver.page_source)

except Exception as e:
    print("An error occurred:", e)
    # Attempt to print the HTML content for troubleshooting, if accessible
    try:
        print("HTML Content for troubleshooting:", driver.page_source)
    except:
        print("Unable to retrieve HTML content.")

finally:
    driver.quit()
```

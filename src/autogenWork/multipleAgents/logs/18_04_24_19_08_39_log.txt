from selenium import webdriver
from selenium.webdriver.chrome.service import Service
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from webdriver_manager.chrome import ChromeDriverManager
import time

# Initialize the WebDriver
driver = webdriver.Chrome(service=Service(ChromeDriverManager().install()))

# Open the website
driver.get("https://privatebin.net")

# Wait for the message textarea to be clickable and enter some text
message_textarea = WebDriverWait(driver, 10).until(
    EC.element_to_be_clickable((By.ID, "message"))
)
message_textarea.send_keys("This is a test post to verify functionality.")

# Select an expiration time for the paste (e.g., 10 minutes)
expiration_dropdown = driver.find_element(By.ID, "expiration")
expiration_dropdown.click()
time.sleep(1)  # Wait for the dropdown to open
expiration_10min = driver.find_element(By.XPATH, "//a[@data-expiration='10min']")
expiration_10min.click()

# Click the "Send" button to create the paste
send_button = driver.find_element(By.ID, "sendbutton")
send_button.click()

# Wait for the paste link to be displayed
paste_link_element = WebDriverWait(driver, 10).until(
    EC.visibility_of_element_located((By.ID, "pastelink"))
)

# Print the paste link
print(paste_link_element.text)

# Print the HTML content of the page to verify the paste was created and to find the link
print(driver.page_source)

# Close the browser
driver.quit()
